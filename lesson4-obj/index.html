cycle `for` for objects

                    
                        for (let key in obj) {
                            console.log(key); // `login`, `pass`
                            console.log(obj[key]) // `solo`, 777
                        }

structuredClone(obj)

simple cloning

                    
                        const obj1 = {code: '007'}
                        const obj2 = obj1
                        obj2.code = '008'
                        console.log(obj1.code); // 008
                        console.log(obj1 === obj2) // true

deep cloning

                    
                        const obj1 = {code: '007'}
                        const obj2 = structuredClone(obj1)
                        obj2.code = '008'
                        console.log(obj1.code); // 007 
                        console.log(obj1 === obj2) // false

new

                    
                        function User(login, pass, role = 'user') {
                            this.login = login
                            this.pass = pass
                            this.role = role
                            this.getInfo = function() {
                                return `login: ${this.login}; role: ${this.role};`
                            }
                        }
                        const user1 = new User('Anna', 1111)
                        const user2 = new User('Bill', 2222)
                        const user3 = new User('Clint', 3333, 'admin')
                        console.log(user2.getInfo()) // login: Bill; role: user;

Optional chaining “?.”

                    
                        const profile = null
                        console.log(profile.name); // error
                        console.log(profile?.name); // undefined
