Switch

                
                    const userName = prompt(`who are you?`, `anonim`)
                    switch (userName) {
                        case `admin`:
                            console.log(`Hello owner`)
                            break
                        case `anonim`:
                            console.log(`Hi, hidden user`)
                            break
                        case null:
                            console.log(`Have no name...`)
                            break
                        default:
                            console.log(`Hello user ${userName}`);
                    }

Function Declaration

Function Expression

                    
                        const sayHi = function(name) {

Arrow function

                    
                        const sayHi = (name) => {
                            return `Hi ${name}`
